function showSuccess(pass,hasGeo){ const code=pass.code||pass.id||'—'; const pub=`https://www.passqr.com/p/${code}`; // Smart redirect URL: auto-detects iOS/Android and opens the right wallet const walletRedirect=`https://dasecure.github.io/passqr-studio/wallet.html?code=${code}`; document.getElementById('result-code').textContent=code; document.getElementById('result-url').textContent=pub; document.getElementById('apple-link').href=`https://www.passqr.com/api/wallet/apple?code=${code}`; document.getElementById('google-link').href=`https://www.passqr.com/api/wallet/google?code=${code}`; document.getElementById('whatsapp-link').href=`https://wa.me/?text=${encodeURIComponent('Your demo pass: '+pub)}`; // QR points to smart wallet redirect — scan on phone → straight to Apple/Google Wallet document.getElementById('qr-img').src=`https://api.qrserver.com/v1/create-qr-code/?size=200x200&margin=10&color=070B14&bgcolor=FFFFFF&data=${encodeURIComponent(walletRedirect)}`; document.getElementById('desktop-hint').style.display=/Mobi|Android|iPhone/i.test(navigator.userAgent)?'none':'block'; document.getElementById('geo-fence-notice').style.display=hasGeo?'block':'none'; document.getElementById('success-modal').classList.add('open'); }
