// ── Images push ─────────────────────────────────────────────────────────────── document.getElementById('push-images-btn').addEventListener('click',async()=>{ if(!images.logo&&!images.icon&&!images.strip){toast('Upload at least one image','error');return;} const btn=document.getElementById('push-images-btn');btn.innerHTML='';btn.disabled=true; try{ // Upload each image type separately to POST /api/template/image const uploads=[ images.logo && apiCall('POST','/api/template/image',{template_id:TEMPLATE_ID,type:'logo', data:images.logo}), images.icon && apiCall('POST','/api/template/image',{template_id:TEMPLATE_ID,type:'icon', data:images.icon}), images.strip && apiCall('POST','/api/template/image',{template_id:TEMPLATE_ID,type:'strip',data:images.strip}), ].filter(Boolean); await Promise.all(uploads); toast('Images pushed ✓','success'); }catch(e){toast('Push failed: '+e.message,'error');} btn.innerHTML='Push to Template →';btn.disabled=false; });
